
Self signed certificate:

https://stackoverflow.com/questions/10175812/how-to-generate-a-self-signed-ssl-certificate-using-openssl

# interactive
openssl req -nodes -x509 -newkey rsa:4096 -keyout key.pem -out cert.pem -sha256 -days 365

# non-interactive and 10 years expiration
openssl req -nodes -x509 -newkey rsa:4096 -keyout star_xx_com.key -out star_xx_com.crt -sha256 -days 3650 -nodes -subj "/C=CO/ST=Bogota/L=Bogota/O=PoliceVR/OU=DIEPO/CN=policevr.com"

cat star_xx_com.crt star_xx_com.key > star_xx_com.pem
